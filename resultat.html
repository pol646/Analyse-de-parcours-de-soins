# ── 1. Chargement ────────────────────────────────────────────
df <- suppressWarnings(read_excel("data/tous_les_parcours_patients.xlsx")) %>%
  mutate(
    date_debut = as.Date(if_else(TYPE == "Hospitalisation", HOS_EXE_SOI_DTD, EXE_SOI_DTD)),
    date_fin   = as.Date(if_else(TYPE == "Hospitalisation", HOS_EXE_SOI_DTF, EXE_SOI_DTF))
  )

# ── 2. Codage de l'état de chaque acte ───────────────────────
# Priorité : H1 > H2 > T > M > I > P > PH > L > S
df <- df %>%
  mutate(etat = case_when(
    TYPE == "Hospitalisation" & str_starts(as.character(GRG_GHM), "05") ~ "H1",  # hospit. cardiaque
    TYPE == "Hospitalisation"                                            ~ "H2",  # hospit. autre
    PRS_NAT_REF %in% c(1192, 1191, 1086)                                ~ "T",   # téléconsultation
    PRS_NAT_REF == 1110                                                  ~ "M",   # médecin généraliste
    PSE_ACT_NAT == 24                                                    ~ "I",   # infirmier
    PSE_ACT_NAT %in% c(26, 21)                                          ~ "P",   # paramédical
    PSE_ACT_NAT == 50                                                    ~ "PH",  # pharmacie
    PSE_ACT_NAT == 30                                                    ~ "L",   # laboratoire
    PSE_SPE_COD %in% c(22, 23, 12, 15, 33)                             ~ "S",   # spécialiste
    TRUE                                                                 ~ NA_character_
  )) %>%
  filter(!is.na(etat))

# ── 3. Origine = 1ère hospitalisation cardiaque (H1) ─────────
df <- df %>%
  group_by(NUM_ENQ) %>%
  filter(any(etat == "H1")) %>%          # garder seulement les patients H1
  mutate(date_origine = min(date_debut[etat == "H1"], na.rm = TRUE)) %>%
  ungroup() %>%
  mutate(semaine = as.integer(floor(as.numeric(date_debut - date_origine) / 7))) %>%
  filter(between(semaine, 0, 51))        # 1 an de suivi (S0–S51)

# ── 4. État par semaine (combinaisons possibles) ─────────────
# Ex. : si H1 et PH la même semaine → "H1+PH"
combiner_etats <- function(etats) {
  etats <- sort(unique(etats))
  if ("H1" %in% etats) return(paste(c("H1", setdiff(etats, "H1")), collapse = "+"))
  if ("H2" %in% etats) return(paste(c("H2", setdiff(etats, "H2")), collapse = "+"))
  paste(etats, collapse = "+")
}

df_semaine <- df %>%
  group_by(NUM_ENQ, semaine) %>%
  summarise(etat_semaine = combiner_etats(etat), .groups = "drop")

# ── 5. Ajout de l'état F (fin de suivi : décès ou sortie) ────
DATE_FIN_SUIVI <- as.Date("2025-01-01")

fin_suivi <- df %>%
  select(NUM_ENQ, date_origine, date_debut, BEN_DCD_DTE) %>%
  group_by(NUM_ENQ) %>%
  summarise(
    date_origine   = first(date_origine),
    derniere_date  = max(date_debut, na.rm = TRUE),
    date_dcd       = {
      d <- as.Date(first(na.omit(BEN_DCD_DTE)))
      if_else(!is.na(d) & d != as.Date("1599-12-31"), d, NA_Date_)
    },
    .groups = "drop"
  ) %>%
  mutate(
    date_fin_patient = if_else(!is.na(date_dcd), date_dcd, derniere_date),
    sem_fin = as.integer(floor(as.numeric(date_fin_patient - date_origine) / 7)) + 1L
  ) %>%
  filter(between(sem_fin, 0, 51), date_fin_patient < DATE_FIN_SUIVI)

# Générer les lignes "F" de sem_fin à S51 pour chaque patient sorti
rows_F <- fin_suivi %>%
  rowwise() %>%
  reframe(NUM_ENQ = NUM_ENQ, semaine = seq(sem_fin, 51L), etat_semaine = "F")

# Fusionner : les actes réels ont la priorité sur F
df_semaine <- df_semaine %>%
  anti_join(rows_F, by = c("NUM_ENQ", "semaine")) %>%
  bind_rows(rows_F) %>%
  arrange(NUM_ENQ, semaine)
# ── 6. Format large : 1 ligne par patient, 1 colonne par semaine ─
col_sem <- paste0("S", 0:51)

df_wide <- df_semaine %>%
  # On s'assure une dernière fois que S0 est H1 dans les données long
  mutate(etat_semaine = if_else(semaine == 0, "H1", etat_semaine)) %>%
  pivot_wider(
    names_from = semaine, 
    names_prefix = "S", 
    values_from = etat_semaine, 
    values_fn = first
  ) %>%
  # Gestion des colonnes manquantes
  { for (col in col_sem) if (!col %in% names(.)) .[[col]] <- NA_character_; . } %>%
  select(NUM_ENQ, all_of(col_sem)) %>%
  # Remplissage des vides par "R"
  mutate(across(all_of(col_sem), ~ replace_na(., "R"))) %>%
  # SÉCURITÉ : On force la colonne S0 à "H1" pour tout le monde
  mutate(S0 = "H1") 

# On identifie les patients qui n'ont QUE "R" ou "F" de S1 à S51
indices_suivi <- apply(df_wide[, paste0("S", 1:51)], 1, function(x) {
  any(!(x %in% c("R", "F")))
})

# On filtre le tableau
df_wide <- df_wide[indices_suivi, ]
